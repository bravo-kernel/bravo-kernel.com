function removeFileString($fileContent, $searchString) {
    $match = $fileContent | Select-String "$searchString" | Select-Object -First 1

    if (-Not($match)) {
        return $fileContent
    }
    $lineNumber = $match.LineNumber - 1

    Write-Host "=> DELETING: $searchString"
    $fileContent.RemoveAt($lineNumber)

    return $fileContent
}


function replaceFileString($fileContent, $searchString, $newValue) {

    $match = $fileContent | Select-String "$searchString" | Select-Object -First 1

    if (-Not($match)) {
        return $fileContent
    }

    $lineNumber = $match.LineNumber - 1

    Write-Host "=> REPLACING: $searchString with $newValue"
    $fileContent[$lineNumber] = $newValue

    return $fileContent
}
